// 1. Include generated header

#include <uORB/topics/my_topic.h>

// 2. Instantiate Publication handle

// Handle automatically advertises topic on construction

uORB::Publication<my_topic_s> _my_topic_pub{ORB_ID(my_topic)};

// 3. Populate and publish (inside Run loop)

void Run() {

my_topic_s msg{};

// Always timestamp your data!

msg.timestamp = hrt_absolute_time();

msg.value = 42.0f;

msg.active = true;

// Publish data

_my_topic_pub.publish(msg);

}

// 1. Include generated header

#include <uORB/topics/my_topic.h>

// 2. Define struct and handle

struct my_topic_s msg;

memset(&msg, 0, sizeof(msg));

orb_advert_t topic_handle = NULL;

// 3. Advertise & Publish loop

msg.timestamp = hrt_absolute_time();

msg.value = 43.0f;

if (topic_handle == NULL) {

// First publication advertises the topic

topic_handle = orb_advertise(ORB_ID(my_topic), &msg);

} else {

// Subsequent publications update it

orb_publish(ORB_ID(my_topic), topic_handle, &msg);

}

Use the C++ wrapper uORB::Publication in modules for RAII safety and cleaner code.
ORB_ID(my_topic) expands to struct pointer
